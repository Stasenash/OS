--5На какую общую сумму были сделаны заказы с марта по май 2010 года?

from MARKET.ORDERS o

--Вывести средний размер комиссионных продавцов для каждого города (указать также город в результате запроса). Отсортировать результаты по убыванию
select s.CITY, AVG(s.COMM)
from MARKET.SALESPEOPLE s
group by s.city
order by s.COMM;

--Вывести число покупателей, закрепленных к каждому из продавцов (указать также имя и город продавца в результате запроса).Отсортировать по именам продавцов в алфавитном порядке
select s.SNAME,s.CITY, count(c.CNUM)
from  MARKET.SALESPEOPLE s join MARKET.CUSTOMERS c on s.SNUM = c.SNUM
group by s.SNAME, s.CITY
order by s.SNAME;

--Найти средний чек (среднюю сумму заказа) для каждого месяца (указать также в результате месяц и год). Отсортировать в порядке следования месяцев в году
select avg(o.AMT), o.ODATE
from MARKET.ORDERS o
group by o.ODATE
order by o.ODATE;

--Вывести максимальный рейтинг покупателя в каждом городе. Не включать в результат запроса города с максимальными рейтингами менее 200
select c.city, max(c.raiting)
from market.customers c
group by c.city
having max(c.raiting)>=200;

--Вывести средний размер заказа покупателей. Включить в результат запроса только покупателей, сделавших более 1 заказа. Отсортировать в порядке возрастания среднего
select avg(o.AMT), c.CNAME
from MARKET.ORDERS o join MARKET.CUSTOMERS c on o.CNUM = c.CNUM
group by c.CNAME
having count(o.CNUM) > 1
order by avg(o.AMT);

--Вывести сумму заказов за каждый месяц 2000 года. Исключить месяцы, в котором было сделано менее 3 заказов. Отсортировать в порядке следования месяцев в году
select trunc(o.ODATE,'mm'), sum(o.AMT)
from MARKET.ORDERS o
group by trunc(o.ODATE,'mm')
having count(*) > 2 --trunc(o.ODATE,'mm')
order by trunc(o.ODATE,'mm');

--Вывести минимальную месячную сумму заказов из всех месяцев в году
select min(sum(o.amt))
from MARKET.ORDERS o
group by o.ODATE;
